**1. 什么是`深拷贝`？如何进行`深拷贝`？**

初始值保存在栈内存中，复杂值保存在堆内存中；如果直接将复杂值去赋值会，前一个值更改时，后一个值也会进行更改

```javascript
const obj = { x: 1, y: 2 };
const tmp = obj;
tmp.x = 2;
console.log(obj.x); // 2
```

深拷贝就是对原对象的数据完全进行拷贝，不涉及数据共享。

```javascript
/*
 * 对于数组
 */
const newArray = oldArray.concat();
const newArray = oldArray.slice();
/*
 * 对于对象
 */
function deepClone(obj) {
  if (typeof obj !== "object") return;
  let result = obj instanceof Array ? [] : {};
  for (let item in obj) {
    if (obj.hasOwnProperty(item)) {
      // 判断是否为自由属性
      result[item] = typeof item === "object" ? deepClone(item) : obj[item];
    }
  }
  return result;
}

/*
 * jsonClone
 * 不能解决循环引用的对象
 * 会忽略 undefined
 * 会忽略 symbol
 * 不能序列化函数
 */
function jsonClone(obj) {
  return JSON.parse(JSON.stringify(obj));
}

/**
 * 使用 MessageChannel
 * 可以拷贝undefined和循环引用的问题
 * 无法拷贝函数
 */
function deepClone(obj) {
  return new Promise((resolve) => {
    const { port1, port2 } = new MessageChannel();
    port2.onmessage = (event) => resolve(event.data);
    port1.postMessage(obj);
  });
}
(async () => {
  const clone = await deepClone(obj);
})();
```

**1. 手动实现`call`**

```javascript
// func.call(obj, ...args);
Function.prototype.myCal = function (context) {
  if (typeof this !== "function") {
    throw new Typeerror();
  }
  // 1. context为可选参数，不传的话默认为windows
  context = context || windows;
  // 2. 给context创建fn属性，其值为需要调用的函数
  context.fn = this;
  // 3. call可以传多个参数，所以需要剥离参数
  const args = [...arguments].slice(1);
  // 4. 调用函数
  const result = context.fn(...args);
  // 5. 删除对象上的函数
  delete context.fn;
  // 6. 返回结果
  return result;
};
```

**2. 手动实现`apply`**

```javascript
// func.apply(obj, array);
Function.prototype.myApply = function (context) {
  if (typeof this !== "function") {
    throw new TypeError();
  }
  context = context || windows;
  context.fn = this;
  let result;
  // 参数处理上与call有所不同
  if (arguments[1]) {
    result = context.fn(...arguments[1]);
  } else {
    result = context.fn();
  }
  delete context.fn;
  return result;
};
```

**3. 手动实现`bind`**

```javascript
// const newFunc = func.bind(obj);
// newFunc();

Function.prototype.myBind = function (context) {
  if (typeof this !== "function") {
    throw new TypeError();
  }
  const _this = this;
  const args = [...arguments].slice(1);
  // bind返回一个函数，故需要考虑函数的两种调用方式
  return function F() {
    if (this instanceof F) {
      // 不会被任意方式改变this，需要忽略传入的this
      return new _this(...args, ...arguments);
    } else {
      // bind可以实现类似f.bind(obj,1)(2)的代码，故需要将两边的参数拼起来
      return _this.apply(context, args.concat(...arguments));
    }
  };
};
```

**3. 什么是防抖？什么是节流**

> 防抖和节流的作用都是防止函数多次调用。区别在于，假设一个用户一直触发这个函数，且每次触发函数的间隔小于 wait，防抖的情况下只会调用一次，而节流的 情况会每隔一定时间（参数 wait）调用函数。

```javascript
/**
 * func是需要防抖的函数，wait是触发事件
 * 低阶防抖，不带立即执行选项
 */
const debounce = (func, wait = 50) => {
  let timer = 0;
  return function (...args) {
    if (timer) clearTimeout(timer);
    timer = setTimeout(() => {
      func.apply(this, args);
    }, wait);
  };
};
/**
 * 带有立即执行选项
 */
function debounce(func, wait = 50, immediate = true) {
  let timer, context, args;

  // 延迟执行函数
  const later = () => {
    timer = null;
    setTimeout(() => {
      if (!immediate) {
        func.apply(context, args);
        context = args = null;
      }
    }, wait);
  };

  return function (...params) {
    if (!timer) {
      timer = later();
      if (immediate) {
        func.apply(this, params);
      } else {
        context = this;
        args = params;
      }
    } else {
      clearTimeout(timer);
      timer = later();
    }
  };
}
```

```html
<div class="container">
  <div class="img">
    <img class="pic" src="" data-src="./img/1.png" />
  </div>
  <div class="img">
    <img class="pic" src="" data-src="./img/2.png" />
  </div>
  <div class="img">
    <img class="pic" src="" data-src="./img/3.png" />
  </div>
  <div class="img">
    <img class="pic" src="" data-src="./img/4.png" />
  </div>
</div>
```

```javascript
// 获取所有图片
const imgs = document.querySelectorAll(".container .img");
// 获取父元素
const container = document.querySelector(".container");
// 获取窗口高度
const viewHeight = window.innerHeight;
const load = lazy();
function lazy() {
  // 运用闭包count进行计数
  let count = 0;
  return function () {
    for (let i = count; i < imgs.length; i++) {
      let distance = imgs[i].getBoundingClientRect().top;
      if (distance >= 0) {
        imgs[i].src = imgs[i].getAttribute("data-src");
        count++;
      }
    }
  };
}
container.addEventListener("scroll", load, false);
```
